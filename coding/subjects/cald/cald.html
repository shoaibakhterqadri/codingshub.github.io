2) Write a program in MIPS assembly language that takes input and display whether number is prime or not.

Output:


Source Code:

.data
input: .asciiz "Enter any number = "
notPrime: .asciiz "This number is not prime number"
prime: .asciiz "This is Prime Number"
.text
.globl main
main:
li $s0,2 #s0 ==> i
la $a0,input
li $v0,4
syscall
li $v0,5
syscall
move $t0,$v0 #t0 user input(n)
blt $t0,$s0,notprime
div $t0,$s0
mflo $t1 #t1 ==> loop n/2
b condition
condition:
ble $s0,$t1,enterLoop
la $a0,prime # this prime number
li $v0,4
syscall
b exit
enterLoop:
div $t0,$s0 #n/i
mfhi $t2 #t2 ==> 0 or 1
beqz $t2,notprime
addi $s0,$s0,1
b condition
notprime:
la $a0,notPrime
li $v0,4
syscall
b exit
exit:
li $v0,10
syscall
